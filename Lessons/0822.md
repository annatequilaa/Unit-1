b = int(input("please enter height b: "))
c = int(input("please enter height c: "))
if a >= b:
    if a >= c:
        if b >= c:
            print(f"{a} {b} {c}")
        else:
            print(f"{a} {c} {b}")
    else:
        print(f"{c} {a} {b}")
elif b >= a: 
  if b >= c:
    if a >= c:
      print(f"{b} {a} {c}")
    else:
      print(f"{b} {c} {a}")
